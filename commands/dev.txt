==========================================================================
 DEVELOPMENT — Docker commands
 Stack: Redis + backend (uvicorn --workers 3) + Celery worker (concurrency 3)
        + frontend + nginx gateway. Same build as prod. Uses AWS RDS (Postgres)
        + AWS S3 (files) via .env — no local db / storage container.
 File:  docker-compose.dev.yml      Env: .env  (from .env.example + DEV profile)
==========================================================================
with start script
./scripts/dev/start.sh

# ---- clear DB data (keeps employees + users) ----
bash scripts/db/delete-pipeline.sh
bash scripts/db/delete-records.sh
# ---- setup (once) ----
./scripts/create-env.sh          # pick LOCAL / DEV / PROD → writes .env
# or: ./scripts/create-env.sh DEV
# then edit DATABASE_URL -> RDS; S3_BUCKET; secrets

# ---- start / stop (or use scripts/dev/start.sh | stop.sh) ----
docker compose -f docker-compose.dev.yml --env-file .env up --build
docker compose -f docker-compose.dev.yml --env-file .env up --build -d   # detached
docker compose -f docker-compose.dev.yml down                                # stop
docker compose -f docker-compose.dev.yml down --volumes                      # stop + wipe local redis

# ---- logs ----
docker compose -f docker-compose.dev.yml logs -f
docker compose -f docker-compose.dev.yml logs -f backend
docker compose -f docker-compose.dev.yml logs -f worker

# ---- status / restart a single service ----
docker compose -f docker-compose.dev.yml ps
docker compose -f docker-compose.dev.yml restart backend
docker compose -f docker-compose.dev.yml build backend

# ---- rebuild frontend ONLY (no backend changes) ----
docker compose -f docker-compose.dev.yml --env-file .env build frontend
docker compose -f docker-compose.dev.yml --env-file .env up -d frontend
docker compose -f docker-compose.dev.yml restart nginx

# ---- shells ----
docker compose -f docker-compose.dev.yml exec backend bash
psql "$DATABASE_URL"   # connect to RDS directly (db is no longer a compose service)

# ---- run the test suite inside the backend container ----
docker compose -f docker-compose.dev.yml exec backend python -m pytest

# ---- one-off Celery worker (if you scale it down) ----
docker compose -f docker-compose.dev.yml exec backend \
  celery -A app.core.celery_app.celery_app worker --loglevel=info --concurrency=1

# ---- URLs ----
# App (SPA + API via nginx):  http://localhost:8080   Login: admin / admin
# On EC2: host nginx + certbot terminate HTTPS in front of :8080 (docs/DEPLOYMENT.md)
